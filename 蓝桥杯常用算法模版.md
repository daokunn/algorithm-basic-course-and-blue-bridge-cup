##### 1.快速乘算法

思路：🌰a * b，可将b转化成2进制，在b转化为十进制时，每位除了乘以权重(2*n)外，还要乘以a，即可得到a * b的结果

```java
import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        boolean flag = false;
        int n = in.nextInt();
        int a = 50, b = 50, val = a;
        int ans = 0, mod = (int)(1e9 + 7);
        while(b != 0){ 
            if(b % 2 == 1){ //如果b为奇数，说明b的二进制最低位是1
                ans += val;
            }
            val *= 2;
            b /= 2;  //等价于b的二进制右移操作
            val %= mod;
            ans %= mod;
        }
        System.out.println(ans);
    }
}
```

##### 7.外卖店优先级

```java
import java.util.*;
public class Main{
    static ArrayList<Integer>[] a = new ArrayList[100010];
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        int N = in.nextInt();
        int M = in.nextInt();
        int T = in.nextInt();
        int[] pro = new int[N + 1];
        int[] last = new int[N + 1];
        HashMap<Integer,Integer>[] arr = new HashMap[T + 1];
        for(int i = 0;i <= T;i ++){
            arr[i] = new HashMap<Integer,Integer>();
        }
        
        for(int i = 0;i < M;i ++){
            int ts = in.nextInt();
            int id = in.nextInt();
            arr[ts].put(id,arr[ts].getOrDefault(id,0) + 1);
        }
        
        HashSet<Integer> ans = new HashSet<>();
        for(int i = 0;i <= T;i ++){
            for(int x : arr[i].keySet()){
                pro[x] = Math.max(0,pro[x] - (i - last[x] - 1));
                if(pro[x] <= 3 && ans.contains(x)) ans.remove(x);
                pro[x] += arr[i].get(x) * 2;
                if(pro[x] > 5 && !ans.contains(x)) ans.add(x);
                last[x] = i;
             }
        }    
        for(int i = 1;i <= N;i ++){
            if(last[i] != T){
                pro[i] = Math.max(0,pro[i] - (T - last[i]));
                if(pro[i] <= 3 && ans.contains(i)) ans.remove(i);
            }
        }
        System.out.println(ans.size());
    }
```

```java
import java.util.*;

        Scanner in = new Scanner(System.in);

        in.nextLine();

        char[] c = s.toCharArray();

            if((i == 0 || c[i - 1] == '.' || c[i - 1] == ' ') && c[i] == 'A' && c[i + 1] == 'l' && c[i + 2] == 'i'
             && c[i + 3] == 'c' && c[i + 4] == 'e' && (c[i + 5] == '.' || c[i + 5] == ' ')){
                 A.add(i);
             }
        }
        for(int i = 0;i < c.length;i ++){
            if((i == 0 || c[i - 1] == '.' || c[i - 1] == ' ') && c[i] == 'B' && c[i + 1] == 'o' && c[i + 2] == 'b'
             && (c[i + 3] == '.' || c[i + 3] == ' ')){
                 B.add(i);
             }
        }
        long ans = 0;
        int cf[] = new int[1000009];
        int qzh[] = new int[1000005];
        for(int t : A){
            cf[Math.max(0,t - k - 3)] ++;
            cf[Math.min(1000000,t + k + 5)] --;
        }

        for(int i = 1;i < 1000005;i ++){
            qzh[i] = qzh[i - 1] + cf[i];
```

##### 11.颜色分割——dfs

```java
import java.util.*;
public class Main{
    static int[][] grid = new int[6][6];
    static boolean[][] s;
    public static int dfs(int x,int y){
        int sum = 1;
        if(x + 1 >= 0 && x + 1 < 6 && !s[x + 1][y] && grid[x][y] == grid[x + 1][y]){
            s[x + 1][y] = true;
            sum += dfs(x + 1,y);
        }
        if(x - 1 >= 0 && x - 1 < 6 && !s[x - 1][y] && grid[x][y] == grid[x - 1][y]){
            s[x - 1][y] = true;
            sum += dfs(x - 1,y);
        }
        if(y + 1 >= 0 && y + 1 < 6 && !s[x][y + 1] && grid[x][y] == grid[x][y + 1]){
            s[x][y + 1] = true;
            sum += dfs(x,y + 1);
        }
        if(y - 1 >= 0 && y - 1 < 6 && !s[x][y - 1] && grid[x][y] == grid[x][y - 1]){
            s[x][y - 1] = true;
            sum += dfs(x,y - 1);
        }
        return sum;
    }
    public static void main(String[] args){
        int ans = 0;
        for(int i = 0;i < Math.pow(2,18);i ++){  //二进制枚举格子状况
            int t = i;
            for(int j = 0;j < 3;j ++){
                for(int k = 0;k < 6;k ++){
                    grid[j][k] = t % 2; 
                    grid[5 - j][5 - k] = 1 - t % 2; //中心对称格子颜色相反
                    t /= 2;  //右移运算
                }
            }
            s = new boolean[6][6];
            s[0][0] = true; //从左上角格子开始dfs
            if(dfs(0,0) == 18) ans ++;
        }
        System.out.println(ans / 4);
    }
}
```

##### 12.全球变暖

```java
import java.util.*;
public class Main {
	public static char[][] c = new char[1010][1010];
	public static int n; 
	public static int[] dx = { -1, 1, 0, 0 };
	public static int[] dy = { 0, 0, -1, 1 };
	public static int[][] overWhelm = new int[1010][1010];//1表示会被淹没，大于1的数表示剩下的小岛
	public static int[][] v = new int[1010][1010];// 判断该陆地是否已经搜索过
	public static long num = 0;// 所有小岛数
	public static Queue<Node> que = new LinkedList<>();

	public static void BFS() {
		int d_num = 2;// 用来表示不同的小岛
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < n; j++) {
				if (c[i][j] == '#' && v[i][j] == 0) {
					Node node = new Node(i, j);
					que.add(node);
					v[i][j] = 1;
					while (!que.isEmpty()) {
						Node tempNode = que.remove();
						check(tempNode.x, tempNode.y, d_num);
						for (int k = 0; k < 4; k++) {
							int tempx = tempNode.x + dx[k];
							int tempy = tempNode.y + dy[k];
							if (tempx >= 0 && tempy >= 0 && tempx < n && tempy < n) {
								if (c[tempx][tempy] == '.') {// 上下左右存在海洋，会被淹没
									continue;
								}
								if (c[tempx][tempy] == '#' && v[tempx][tempy] == 0) {
									que.add(new Node(tempx, tempy));
									v[tempx][tempy] = 1;
								}
							}
						}

					}
					num++;
					d_num++;
				}
			}
		}
	}

	// 判断上下左右是否有海洋
	public static void check(int x, int y, int cur) {
		for (int k = 0; k < 4; k++) {
			int nx = x + dx[k];
			int ny = y + dy[k];
			// 边界之外全是海洋
			if (nx < 0 || ny < 0 || nx >= n || ny >= n) {
				overWhelm[x][y] = 1;
				break;
			}
			if (c[nx][ny] == '.') {
				overWhelm[x][y] = 1;
				break;
			}
			overWhelm[x][y] = cur;
		}
	}

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		n = sc.nextInt();
		for (int i = 0; i < n; i++)
			c[i]=sc.next().toCharArray();
		BFS();
		HashSet<Integer> set = new HashSet<>();// 存放的是未被淹没的小岛
		for (int i = 0; i < n; i++)
			for (int j = 0; j < n; j++)
				if (overWhelm[i][j] > 1)
					set.add(overWhelm[i][j]);
		System.out.println(num - set.size());
	}
	static class Node {
		int x;
		int y;
		
		Node(int x, int y) {
			this.x = x;
			this.y = y;
		}
	}
}
```
