##### 1.1单链表(静态)

```java
import java.util.*;
public class Main{
    static int[] val = new int[100010]; //存放结点值
    static int[] ne = new int[100010];  //存放结点的next指针 
    static int head = -1;  //初始化头结点的下一个结点为空结点，用-1表示
    static int idx = 0; //记录当前用到哪个点,每次idx加一相当于new一个新结点
    
    //头插法
    public static void insertToHead(int x){
        val[idx] = x;
        ne[idx] = head;
        head = idx;
        idx ++;
    }
    
    //在下标为k的结点后插入一个节点
    public static void insert(int k,int x){
        ne[idx] = ne[k];
        val[idx] = x;
        ne[k] = idx;
        idx ++;
    }
    
    //删除下标为k的节点的后一个节点
    public static void delete(int k){
        ne[k] = ne[ne[k]];
    }
    
    public static void main(String[] args){
        Scanner in = new Scanner(System.in);
        int T = in.nextInt();
        while(T -- != 0){
            char opr = in.next().charAt(0);
            if(opr == 'H'){
                int x = in.nextInt();
                insertToHead(x);
            }
            else if(opr == 'I'){
                int k = in.nextInt();
                int x = in.nextInt();
                insert(k - 1,x);
            }else{
                int k = in.nextInt();
                if(k == 0) head = ne[head];
                else delete(k - 1);
            }
        }
        
        for(int i = head;i != -1;i = ne[i]) System.out.print(val[i] + " ");
        
    }
}
```

##### 1.2.双链表(静态)

```java
import java.util.*;
public class Main{
    
    static int[] l = new int[100010];
    static int[] r = new int[100010];
    static int[] val = new int[100010];
    static int idx = 0;

    //在下标为k的结点的右边插入一个结点
    public static void insert(int k,int x){
        val[idx] = x;
        l[idx] = k;
        r[idx] = r[k];
        l[r[k]] = idx;
        r[k] = idx;
        idx ++;
    }
    
    //删除下标为k的结点
    public static void delete(int k){
        l[r[k]] = l[k];
        r[l[k]] = r[k];
    }

    public static void main(String[] args){
        //初始化左端点为0，右端点为1
        r[0] = 1; //0的右边为1
        l[1] = 0; //1的左边为0
        idx  = 2; //由于初始化左右端点，所以idx从2开始
        
        Scanner in = new Scanner(System.in);
        int T = in.nextInt();
        while(T -- != 0){
            String opr = in.next();
            if(opr.equals("L")){
                int x = in.nextInt();
                insert(0,x);
            }else if(opr.equals("R")){
                int x = in.nextInt();
                insert(l[1],x);
            }else if(opr.equals("D")){
                int k = in.nextInt();
                delete(k + 1);
            }else if(opr.equals("IL")){
                int k = in.nextInt();
                int x = in.nextInt();
                insert(l[k + 1],x);
            }else{
                int k = in.nextInt();
                int x = in.nextInt();
                insert(k + 1,x);
            }
        }
        
        for(int i = r[0];i != 1;i = r[i]) System.out.print(val[i] + " ");
        
    }
}
```
